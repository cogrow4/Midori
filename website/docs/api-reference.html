API Reference

Built-in functions and types available in every Midori program.

Output

Functions

  • print(s: Str) -> Nil
  • println(s: Str) -> Nil

print(s: Str) -> Nil

Print a string to stdout without a trailing newline.

println(s: Str) -> Nil

Print a string to stdout followed by a newline.

fn main() {
    print("hello")    // no newline
    println("world")  // with newline
}
// Output: helloworld

Type Conversion

str(n: Int) -> Str

Convert an integer to its string representation.

str_float(f: Float) -> Str

Convert a float to its string representation.

str_bool(b: Bool) -> Str

Convert a boolean to "true" or "false".

str_char(c: Char) -> Str

Convert a character to a single-character string.

int(s: Str) -> Int

Parse a string as an integer. Returns 0 on invalid input.

let n = int("42")     // n is 42 (Int)
let s1 = str(n)        // s1 is "42"
let s2 = str_float(3.14)  // s2 is "3.14"
let s3 = str_bool(true)   // s3 is "true"

String Operations

len_str(s: Str) -> Int

Return the length of a string in bytes.

s.len() -> Int (method)

String method — returns the length in bytes.

s.is_empty() -> Bool (method)

String method — returns true if the string length is 0.

let s = "hello"
let n = len_str(s)     // 5
let n2 = s.len()       // 5 (method dispatch)
let empty = s.is_empty()  // false

String Indexing

Access individual characters by index: s[i] returns a Char at byte position i.

let s = "hello"
let c = s[0]  // 'h'

String Concatenation

Use + to concatenate strings:

let s = "hello" + " " + "world"  // "hello world"

String Interpolation

Embed expressions inside strings with { }:

let name = "Midori"
let year = 2026
println("Welcome to {name}, version {year}!")
// Output: Welcome to Midori, version 2026!

Array Operations

len(arr: Array) -> Int

Return the number of elements in an array.

arr.len() -> Int (method)

Array method — returns the number of elements.

arr.is_empty() -> Bool (method)

Array method — returns true if the array has 0 elements.

let arr = [1, 2, 3]
let n = len(arr)     // 3
let n2 = arr.len()   // 3 (method dispatch)

File I/O

read_file(path: Str) -> Str

Read the entire contents of a file. Returns an empty string if the file doesn't exist.

write_file(path: Str, content: Str) -> Bool

Write a string to a file. Returns true on success, false on failure.

file_exists(path: Str) -> Bool

Returns true if a file exists at the given path.

let content = read_file("test.txt")
let ok = write_file("output.txt", "hello")
let exists = file_exists("test.txt")

CLI Arguments

os_args() -> Array(Str)

Returns an array of command-line arguments passed to the program. The first argument (args[0]) is the program name.

let args = os_args()
let prog_name = args[0]
for arg in args {
    println(arg)
}

Ranges

range(n: Int) -> Array(Int)

Create an array of integers from 0 to n-1.

range(start: Int, end: Int) -> Array(Int)

Create an array of integers from start to end-1.

for i in range(5) {
    println(str(i))
}
// Prints 0, 1, 2, 3, 4

for i in range(2, 5) {
    println(str(i))
}
// Prints 2, 3, 4

Note: range returns an Array, so it works with for...in loops and len().

String Builder

For efficient string construction:

sb_new() -> StrBuilder

Create a new string builder.

sb_append(builder, s: Str) -> Nil

Append a string to the builder.

sb_build(builder) -> Str

Build the final string from the builder contents.

let sb = sb_new()
sb_append(sb, "hello")
sb_append(sb, " ")
sb_append(sb, "world")
let result = sb_build(sb)  // "hello world"
